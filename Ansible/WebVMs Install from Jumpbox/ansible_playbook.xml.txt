---
- name: Config Web VM with Docker
  hosts: webservers
  become: true
  tasks:
  - name: Install program docker.io
    apt:
      force_apt_get: yes
      update_cache: yes
      name: docker.io
      state: present

  - name: Install program python3-pip
    apt:
      force_apt_get: yes
      name: python3-pip
      state: present

  - name: Docker
    pip:
       name: docker
       state: present

  - name: Cyberxsecurity/dvwa
    docker_container:
      name: dvwa
      image: cyberxsecurity/dvwa
      state: started
      published_ports: 80:80

  - name: Vm start
    systemd:
       name: docker
       state: started
       enabled: yes
...
